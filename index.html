Ejercicio 1: Conversión de millas a km/h y galones a litros

InputVelocidad en millas/hora, capacidad en galones
Proceso1 mph = 1.60934 km/h; 1 galón = 3.78541 litros
OutputVelocidad en km/h y capacidad en litros
#include <iostream>
using namespace std;
int main() {
    double mph, gal;
    cout << "Ingrese velocidad en mph: ";
    cin >> mph;
    cout << "Ingrese capacidad en galones: ";
    cin >> gal;
    double kmh = mph * 1.60934;
    double litros = gal * 3.78541;
    cout << "Velocidad en km/h: " << kmh << endl;
    cout << "Capacidad en litros: " << litros << endl;
    return 0;
}

Ejercicio 2: Costo de envío

InputPeso (kg), distancia (km)
ProcesoCosto base: 5000 * peso. Cada 100 km = +10%
OutputCosto total de envío
#include <iostream>
using namespace std;
int main() {
    double peso, distancia;
    cout << "Peso en kg: ";
    cin >> peso;
    cout << "Distancia en km: ";
    cin >> distancia;
    double costo = peso * 5000;
    int tramos = distancia / 100;
    costo += costo * 0.10 * tramos;
    cout << "Costo total: $" << costo << endl;
    return 0;
}

Ejercicio 9: Conversión de nudos a km/h

InputVelocidad en nudos
Proceso1 nudo = 1.852 km/h; rango embarcación
OutputVelocidad en km/h + tipo embarcación
#include <iostream>
using namespace std;
int main() {
    double nudos;
    cout << "Velocidad en nudos: ";
    cin >> nudos;
    double kmh = nudos * 1.852;
    cout << "Velocidad en km/h: " << kmh << endl;
    if(kmh < 20) cout << "Tipo: Pesquero" << endl;
    else if(kmh < 50) cout << "Tipo: Carguero" << endl;
    else if(kmh < 100) cout << "Tipo: Yate" << endl;
    else cout << "Velocidad fuera de rango definido" << endl;
    return 0;
}

Ejercicio 10: Consumo de agua mensual

InputNúmero de personas, consumo real (litros)
Procesoconsumo_recomendado = personas * 50 * 30
OutputComparación consumo real vs recomendado
#include <iostream>
using namespace std;
int main() {
    int personas;
    double consumo;
    cout << "Personas en el hogar: ";
    cin >> personas;
    cout << "Consumo real mensual (litros): ";
    cin >> consumo;
    double recomendado = personas * 50 * 30;
    cout << "Consumo recomendado: " << recomendado << " litros" << endl;
    cout << "Consumo real: " << consumo << " litros" << endl;
    if(consumo <= recomendado)
        cout << "Consumo adecuado" << endl;
    else
        cout << "Consumo excesivo" << endl;
    return 0;
}
